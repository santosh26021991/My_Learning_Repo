**Stored procedure: is set of sql statements

create or replace procedure Transfer_money(
	IN from_acc INT,
	IN to_acc INT,
	IN amount INT)
	
	BEGIN
		UPDATE accounts set balance = balance - amount where acc_no = from_acc;
		UPDATE accounts set balance = balance + amount where acc_no = to_acc;
		INSERT into transactions values (from_acc, to_acc, amount);
	END;
